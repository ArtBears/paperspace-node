machines/restart.js

'use strict';

var method = require('./../method');
var assign = require('lodash.assign');

/**
 * @memberof machines
 * @method restart
 * @description Restart an individual machine. If the machine is already restarting,
 * this action will request the machine be restarted again.
 * This action can only be performed by the user who owns the machine.
 * @param {object} params - Machine restart parameters
 * @param {string} params.publicMachineId - Id of the machine to shut down
 * @param {function} cb - Node-style error-first callback function
 * @example
 * paperspace.machines.restart({
 *   publicMachineId: psabcdef,
 * }, function(err, resp) {
 *   // handle error or http response
 * });
 * @example
 * $ paperspace machines restart --publicMachineId psabcdef
 * @example
 * # HTTP request:
 * POST /machines/restart
 * # Returns 200 on success
 */

function restart(params, cb) {
	return method(restart, params, cb);
}

assign(restart, {
	auth: true,
	group: 'machines',
	name: 'restart',
	method: 'post',
	route: '/machines/restart',
	requires: {
		publicMachineId: 'string',
	},
	returns: {},
});

module.exports = restart;
